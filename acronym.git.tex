% git commands as acronyms

\usepackage{xcolor}
\snippet{acronym}

\definecolor{gitcommand}{gray}{0.45}
\newcommand{\gitcommandhelper}[1]{\textcolor{gitcommand}{\texttt{#1}}}
\newcommand{\gitcommand}[2]{%
	\acronym{#1}%
		{\href{#2}{\gitcommandhelper{#1}} (\url{#2})}%
		{\href{#2}{\gitcommandhelper{#1}}}%
}

% Provides gray header and footer with the section title centered in the header
% and the documenttitle in the bottom
\usepackage{xcolor}
\usepackage{fancyhdr}

\snippet{documenttitle}

% Set header and footer settings
\definecolor{header}{gray}{0.4}
\definecolor{footer}{gray}{0.4}
\pagestyle{fancy}
\fancyhf{}
\fancyhead[C]{\textcolor{header}{\leftmark}}
\fancyfoot[L]{\textcolor{footer}{\documenttitle}}
\fancyfoot[R]{\textcolor{footer}{\thepage}}
%TWOSIDED \fancyfoot[LO,RE]{\documenttitle} % for two sided alternation
%TWOSIDED \fancyfoot[LE,RO]{\thepage} % of position
\renewcommand{\headrulewidth}{0pt}
\renewcommand{\footrulewidth}{0pt}
